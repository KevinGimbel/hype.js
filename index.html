HYPE.js

Hyperscript implementation

Everything in the output tab is generated with the following Code.


const Tags = ['div', 'p', 'link', 'script', 'h1', 'ul', 'li', 'a'];
const sstyle = {style: 'font-size: .95rem; color: red; border-bottom: 2px solid #ddd;'}
const h = new Hype();
h.createTags(Tags);
const content = [
h1('Hello world!'),
h.hy('h2', {}, 'Single Item rendering with h.hy'),
div('Nested rendering with div() function', [
    p('Text rendering with the p() function.'),
    p('Another line!')
              ]),
div('AAAAND we have a list!', [
    ul([
        li('This is a list item.'),
        li('Aaaand another list item.'),
        li('We could do this forever :) .'),
    ])
]),
div('Creating Links with Attributes.',[
    a('Click here!', {
        href: 'https://codepen.com/',
        title: 'CodePen - HTML Playground',
        onClick: "alert('YAY!')"
    })
  ])
];
const app = document.getElementById('app');
h.append(content, app);
